CREATE ENVIRONMENT -- Ctr + Shift + P
    Type: python create Environment
    cmd: conda create -n myenv python=3.13
ACTIVATE ENVIRONMENT
    conda activate myenv
SELECT INTERPETER/ENVIRONMENT -- Miniconda
    Ctrl + Shift + P
    Type: 'Python: Select Interpreter'
    Select: conda (C...\.venv) --- Recommended

INSTALL PACKAGES
General rule	conda install flask pillow sqlalchemy
If 100% sure pip points to the right env (optional shortcut) -- Inside conda
        pip install <package_name>


CHECK ENVIRONMENT
    where python
        .venv --- pip/venv environment.
        miniconda3 --- Conda environment.
        Python313 --- global Python.
